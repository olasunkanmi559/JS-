let age =65
function agecal (retire, dateofbirth){
return dateofbirth-retire
}

let a = prompt('What year were you born?')
let thisyear= new Date().getFullYear();
alert(`You are ${agecal(a,thisyear)} years old`);
if(age>agecal(a,thisyear)){
alert(`You should retire in ${age - agecal(a,thisyear)} years`)
}else if(age==agecal(a,thisyear)){
alert('You can retire now')
}else{
alert(`You are too old for this job retire immediately`)
}
